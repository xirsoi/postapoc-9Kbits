university_yale = {

	society_ranks_gfx = generic_ranks

	}

		has_dlc = "Mystics"

		hidden_trigger = {
			b_new_haven_CT = {
					holder_scope = { NOT = { war_with = ROOT } }
					location = { NOT = { distance = { where = ROOT distance = 400 } } }
			}
			NAND = {
				ai = yes
				is_society_rank_full = { 
					society = university_yale
					rank = 1
				}
			}
			OR = {
				is_landed = yes
				is_patrician = yes
			}
		}
		trait = graduate
		learning = 10		
		NOR = {
			is_nomadic = yes
			is_tribal = yes
			religion_group = pagan_group
			religion_group = pirate
		}
		hidden_trigger = { mercenary = no }
		age = 25
		hidden_trigger = {
			NOT = { has_character_flag = society_join_block }
		}
